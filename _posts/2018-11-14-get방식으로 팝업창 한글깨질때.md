    ex ) jsp?userName="+ escape("유저이름")+" <br/>
    그리고 받는부분
  </p>
  <br/>
  <pre>
    <code>
      var user = "";
      user = location.search.split("&")[?];
	    user = unescape(user.split("=")[?]);
    </code>
  </pre>
